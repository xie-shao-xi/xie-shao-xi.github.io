  3. exec执行还是能执行的，不过我这里用的hexo s，所以需要考虑关闭程序的问题，用exec没办法，查到spawn能获取执行的程序的pid，然后就改成用spawn

    let hexoServer;
const startServer = () => {
    return new Promise((resolve, reject) => {
        const cmd = 'hexo s';
        const cwd = 'D:\\project\\blog';
        let newHexoServer = spawn(cmd, [], {
            cwd,
            // 不设置这个，在window上无法执行
            shell: process.platform === 'win32'
        })
        newHexoServer.stdout.on('data', (data) => {
            let res = JSON.stringify(data.toString())
            console.log("stdout.on" + res);
            if(res.includes("has been used")) {
                reject(res)
            }else {
                hexoServer = newHexoServer;
                resolve("start success!")
            }
        });

        newHexoServer.stderr.on('data', (data) => {
            console.error(JSON.stringify(data.toString()));
            reject("start error,stderr!" + JSON.stringify(data))
        });

        newHexoServer.on('error', (err) => {
            console.error(JSON.stringify(err));
            reject("start error,onerror!" + JSON.stringify(err))
        });

        newHexoServer.on('exit', (code) => {
            console.log(`Child exited with code ${code}`);
        });
    })
}

  4. 然后发现一个问题，stdout.on回调是执行多次的，程序执行后一行打印一次，想了个笨方法，用定时器记录，然后记录过程判断，是不是出现了成功的文案，超过1秒没有继续则默认失败

    const startServer = () => {
    return new Promise((resolve, reject) => {
        const cmd = 'hexo s';
        const cwd = 'D:\\project\\blog';
        let newHexoServer = spawn(cmd, [], {
            cwd,
            shell: process.platform === 'win32'
        })


        // 记录打印文本和倒计时
        let allOut = "", outTimer;
        newHexoServer.stdout.on('data', (data) => {
            allOut += data.toString();

            clearTimeout(outTimer);
            // 若出现成功的标识文本，则结束并返回
            if(allOut.includes("Hexo is running at")) {
                console.log(allOut)
                // 记录程序信息，方便后续关闭
                hexoServer = newHexoServer;
                resolve("start success!")
            }else {
                outTimer = setTimeout(() => {
                    reject(allOut)
                }, 1000)
            }

        });

        // 记录失败打印文本和失败倒计时
        let allErrOut = "", outErrTimer;
        newHexoServer.stderr.on('data', (data) => {
            allErrOut += data.toString();

            clearTimeout(outErrTimer);
            outErrTimer = setTimeout(() => {
                reject(allErrOut)
            }, 2000)
        });

        newHexoServer.on('error', (err) => {
            console.error(JSON.stringify(err));
            reject("start error,onerror!" + JSON.stringify(err))
        });

        newHexoServer.on('exit', (code) => {
            console.log(`Child exited with code ${code}`);
        });

    })
}

const stopServer = () => {
    if(!hexoServer) return Promise.resolve("had closed");
    
    return new Promise((resolve, reject) => {
        let pid = hexoServer.pid;
        exec(`taskkill /PID ${pid} /T /F`, { windowsHide: false }, (error, stdout, stderr) => {
            if(error) {
                console.log("error: " + JSON.stringify(error))
                reject("close fail!")
            }else if(stderr) {
                console.log("stderr: " + stderr)
                reject("close fail!")
            }else {
                resolve("close success!" + stdout)
            }
            hexoServer = undefined;
        });
    })
}
